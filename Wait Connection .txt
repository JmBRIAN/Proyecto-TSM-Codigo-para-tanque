#include <WiFi.h>


WiFiServer server(8888);
WiFiClient client;


void setup() {
  Serial.begin(115200);
  WiFi.softAP("Tanque_WIFI", "12345678");
  server.begin();
  Serial.println("Esperando conexión...");
}


void loop() {
  if (!client || !client.connected()) {
    client = server.available();
  }


  if (client && client.connected()) {
    Serial.println("Cliente conectado");
  }


  delay(500);
}
