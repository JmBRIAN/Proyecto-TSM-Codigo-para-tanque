#include <WiFi.h>


const char* ssid = "Tanque_WIFI";
const char* password = "12345678";


void setup() {
  Serial.begin(115200);


  // Inicia el punto de acceso (AP)
  WiFi.softAP(ssid, password);


  IPAddress IP = WiFi.softAPIP();
  Serial.print("AP configurado. IP: ");
  Serial.println(IP);
}


void loop() {
  delay(1000);
}
